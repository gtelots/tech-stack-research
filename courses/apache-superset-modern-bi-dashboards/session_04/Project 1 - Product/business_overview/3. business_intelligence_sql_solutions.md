# SQL Solutions for Business Intelligence Questions

This document provides SQL queries to answer the business intelligence questions listed in the companion document. Each solution is numbered to correspond with the matching question.

## Product Analysis Solutions

### Basic Product Metrics

**1. What are the top 10 products by total revenue?**
```sql
SELECT 
    p.product_id,
    p.name,
    p.category,
    p.is_active,
    SUM(o.quantity * o.unit_price) AS total_revenue
FROM 
    products p
JOIN 
    orders o ON p.product_id = o.product_id
WHERE 
    o.status != 'Cancelled' AND o.is_returned = FALSE
GROUP BY 
    p.product_id, p.name, p.category
ORDER BY 
    total_revenue DESC
LIMIT 10;
```

**10. What is the sell-through rate for each product?**
```sql
WITH product_sales AS (
    SELECT 
        p.product_id,
        p.name,
        p.category,
        p.stock_quantity AS current_stock,
        COALESCE(SUM(o.quantity), 0) AS total_sold
    FROM 
        products p
    LEFT JOIN 
        orders o ON p.product_id = o.product_id AND o.status != 'Cancelled' AND o.is_returned = FALSE
    GROUP BY 
        p.product_id, p.name, p.category, p.stock_quantity
)
SELECT 
    product_id,
    name,
    category,
    current_stock,
    total_sold,
    CASE 
        WHEN (current_stock + total_sold) = 0 THEN 0
        ELSE (total_sold::float / (current_stock + total_sold)) * 100 
    END AS sell_through_rate
FROM 
    product_sales
ORDER BY 
    sell_through_rate DESC;
```

**17. Which product categories show the highest growth rate in sales?**
```sql
WITH monthly_sales AS (
    SELECT 
        DATE_TRUNC('month', o.order_date) AS month,
        p.category,
        SUM(o.quantity * o.unit_price) AS monthly_revenue
    FROM 
        products p
    JOIN

        AND o.order_date >= CURRENT_DATE - INTERVAL '1 year'
    GROUP BY 
        DATE_TRUNC('month', o.order_date), p.category
),
category_growth AS (
    SELECT 
        category,
        FIRST_VALUE(monthly_revenue) OVER (PARTITION BY category ORDER BY month) AS first_month_revenue,
        LAST_VALUE(monthly_revenue) OVER (PARTITION BY category ORDER BY month ROWS BETWEEN UNBOUNDED PRECEDING AND UNBOUNDED FOLLOWING) AS last_month_revenue
    FROM 
        monthly_sales
)
SELECT 
    category,
    first_month_revenue,
    last_month_revenue,
    (last_month_revenue - first_month_revenue) AS absolute_growth,
    CASE 
        WHEN first_month_revenue = 0 THEN NULL
        ELSE ((last_month_revenue - first_month_revenue) / first_month_revenue * 100)
    END AS growth_percentage
FROM 
    category_growth
GROUP BY 
    category, first_month_revenue, last_month_revenue
ORDER BY 
    growth_percentage DESC NULLS LAST;
```

**20. Which product categories have the most consistent sales throughout the year?**
```sql
WITH monthly_category_sales AS (

        AND o.order_date >= CURRENT_DATE - INTERVAL '1 year'

    SELECT 
        category,
        AVG(monthly_revenue) AS avg_monthly_revenue,
        STDDEV(monthly_revenue) AS stddev_monthly_revenue
    FROM 
        monthly_category_sales
    GROUP BY 
        category
)
SELECT 
    category,
    avg_monthly_revenue,
    stddev_monthly_revenue,
    (stddev_monthly_revenue / NULLIF(avg_monthly_revenue, 0) * 100) AS coefficient_of_variation
FROM 
    category_stats
ORDER BY 
    coefficient_of_variation;
```

### Product Performance

**21. Which products have experienced the greatest increase in sales over the past 3 months?**
```sql
WITH monthly_product_sales AS (
    SELECT 
        p.product_id,
        p.name,
        p.category,
        DATE_TRUNC('month', o.order_date) AS month,
        SUM(o.quantity) AS monthly_units_sold,
        SUM(o.quantity * o.unit_price) AS monthly_revenue
    FROM 
        products p
    JOIN 
        orders o ON p.product_id = o.product_id

        AND o.order_date >= CURRENT_DATE - INTERVAL '3 months'
    GROUP BY 
        p.product_id, p.name, p.category, DATE_TRUNC('month', o.order_date)
),
product_growth AS (
    SELECT 
        product_id,
        name,
        category,
        FIRST_VALUE(monthly_revenue) OVER (PARTITION BY product_id ORDER BY month) AS first_month_revenue,
        LAST_VALUE(monthly_revenue) OVER (PARTITION BY product_id ORDER BY month ROWS BETWEEN UNBOUNDED PRECEDING AND UNBOUNDED FOLLOWING) AS last_month_revenue
    FROM 
        monthly_product_sales
)
SELECT 
    product_id,
    name,
    category,
    first_month_revenue,
    last_month_revenue,
    (last_month_revenue - first_month_revenue) AS absolute_growth,
    CASE 
        WHEN first_month_revenue = 0 THEN NULL
        ELSE ((last_month_revenue - first_month_revenue) / first_month_revenue * 100)
    END AS growth_percentage
FROM 
    product_growth
GROUP BY 
    product_id, name, category, first_month_revenue, last_month_revenue
HAVING 
    first_month_revenue > 0

    growth_percentage DESC NULLS LAST
```

**24. How does discounting affect sales volume for different products?**
```sql
SELECT 
    p.product_id,
    p.name,
    p.category,
    AVG(CASE WHEN o.discount_percentage = 0 THEN o.quantity ELSE NULL END) AS avg_quantity_no_discount,
    AVG(CASE WHEN o.discount_percentage > 0 THEN o.quantity ELSE NULL END) AS avg_quantity_with_discount,
    COUNT(CASE WHEN o.discount_percentage = 0 THEN o.order_id END) AS orders_no_discount,
    COUNT(CASE WHEN o.discount_percentage > 0 THEN o.order_id END) AS orders_with_discount,
    CASE 
        WHEN AVG(CASE WHEN o.discount_percentage = 0 THEN o.quantity ELSE NULL END) = 0 THEN NULL
        ELSE (AVG(CASE WHEN o.discount_percentage > 0 THEN o.quantity ELSE NULL END) - 
              AVG(CASE WHEN o.discount_percentage = 0 THEN o.quantity ELSE NULL END)) / 
              AVG(CASE WHEN o.discount_percentage = 0 THEN o.quantity ELSE NULL END) * 100
    END AS pct_increase_with_discount
FROM 
    products p
JOIN 
    orders o ON p.product_id = o.product_id
WHERE 
    o.status != 'Cancelled' AND o.is_returned = FALSE
GROUP BY 
    p.product_id, p.name, p.category
HAVING 
    COUNT(CASE WHEN o.discount_percentage = 0 THEN o.order_id END) > 10 AND
    COUNT(CASE WHEN o.discount_percentage > 0 THEN o.order_id END) > 10
ORDER BY 
    pct_increase_with_discount DESC NULLS LAST;
```

**28. Which products are frequently ordered together?**
```sql
-- This query would work best with a more normalized orders schema where
-- multiple products can be part of a single order.
-- Here's a conceptual solution for the current schema:

WITH customer_purchases AS (
    SELECT 
        o1.customer_id,
        o1.product_id AS product1,
        o2.product_id AS product2
    FROM 
        orders o1
    JOIN 
        orders o2 ON o1.customer_id = o2.customer_id AND 
                    o1.product_id < o2.product_id AND
                    ABS(EXTRACT(EPOCH FROM (o1.order_date - o2.order_date)) / 86400) <= 7
    WHERE 
        o1.status != 'Cancelled' AND o1.is_returned = FALSE AND
        o2.status != 'Cancelled' AND o2.is_returned = FALSE
)
SELECT 
    p1.name AS product1_name,
    p2.name AS product2_name,
    COUNT(*) AS purchase_pair_count
FROM 
    customer_purchases cp
JOIN 
    products p1 ON cp.product1 = p1.product_id
JOIN 
    products p2 ON cp.product2 = p2.product_id
GROUP BY 
    p1.name, p2.name
ORDER BY 
    purchase_pair_count DESC
LIMIT 
    20;
```

**75. What is our customer retention rate?**
```sql
-- This requires order data across multiple time periods
WITH customer_orders AS (
    SELECT 
        o.customer_id,
        MIN(DATE_TRUNC('month', o.order_date)) AS first_purchase_month,
        MAX(DATE_TRUNC('month', o.order_date)) AS last_purchase_month,
        COUNT(DISTINCT DATE_TRUNC('month', o.order_date)) AS active_months,
        COUNT(DISTINCT o.order_id) AS total_orders
    FROM 
        orders o
    WHERE 
        o.status != 'Cancelled'
    GROUP BY 
        o.customer_id
),
monthly_cohorts AS (
    SELECT 
        first_purchase_month,
        COUNT(DISTINCT customer_id) AS new_customers
    FROM 
        customer_orders
    GROUP BY 
        first_purchase_month
),
monthly_retention AS (
    SELECT 
        co.first_purchase_month,
        DATE_TRUNC('month', o.order_date) AS activity_month,
        COUNT(DISTINCT o.customer_id) AS active_customers
    FROM 
        customer_orders co
    JOIN 
        orders o ON co.customer_id = o.customer_id AND 
                  DATE_TRUNC('month', o.order_date) > co.first_purchase_month
    WHERE 
        o.status != 'Cancelled'
    GROUP BY 
        co.first_purchase_month, DATE_TRUNC('month', o.order_date)
)
SELECT 
    mc.first_purchase_month,
    mc.new_customers,
    mr.activity_month,
    mr.active_customers,
    (mr.active_customers * 100.0 / mc.new_customers) AS retention_rate
FROM 
    monthly_cohorts mc
LEFT JOIN 
    monthly_retention mr ON mc.first_purchase_month = mr.first_purchase_month
ORDER BY 
    mc.first_purchase_month, mr.activity_month;
```

```sql
    CASE 
        WHEN first_month_revenue = 0 THEN NULL
        ELSE ((last_month_revenue - first_month_revenue) / first_month_revenue * 100)
    END AS growth_percentage
FROM 
    channel_growth
GROUP BY 
    sales_channel, first_month_revenue, last_month_revenue
ORDER BY 
    growth_percentage DESC NULLS LAST;
```

**120. What is the average customer lifetime value by acquisition channel?**
```sql
WITH first_order_channel AS (
    SELECT 
        customer_id,
        sales_channel AS first_channel,
        MIN(order_date) AS first_order_date
    FROM 
        orders
    WHERE 
        status != 'Cancelled'
    GROUP BY 
        customer_id
),
customer_lifetime_value AS (
    SELECT 
        foc.customer_id,
        foc.first_channel,
        SUM(o.total_amount) AS total_lifetime_value,
        COUNT(*) AS total_orders
    FROM 
        first_order_channel foc
    JOIN 
        orders o ON foc.customer_id = o.customer_id
    WHERE 
        o.status != 'Cancelled'
    GROUP BY 
        foc.customer_id, foc.first_channel
)
SELECT 
    first_channel,
    COUNT(*) AS customer_count,
    AVG(total_lifetime_value) AS avg_lifetime_value,
    PERCENTILE_CONT(0.5) WITHIN GROUP (ORDER BY total_lifetime_value) AS median_lifetime_value,
    AVG(total_orders) AS avg_lifetime_orders
FROM 
    customer_lifetime_value
GROUP BY 
    first_channel
ORDER BY 
    avg_lifetime_value DESC;
```
